import express from "express";
import fetch from "node-fetch";

const app = express();
app.use(express.json());

app.post("/tasa", async (req, res) => {
  try {
    // Obtener tasa BCV
    const resp = await fetch("https://pydolarvenezuela-api.vercel.app/api/v1/dollar");
    const data = await resp.json();

    const tasaBCV = data.monitors.bcv.price;

    // Datos que envÃ­a Builderbot
    const { totalUSD } = req.body;

    // Calcular en Bs
    const totalBs = totalUSD * tasaBCV;

    res.json({
      tasa_bcv: tasaBCV.toFixed(2),
      total_usd: totalUSD.toFixed(2),
      total_bs: totalBs.toFixed(2),
      mensaje: `Total: $${totalUSD.toFixed(2)} | Bs. ${totalBs.toFixed(2)} (Tasa BCV: ${tasaBCV.toFixed(2)})`
    });

  } catch (error) {
    console.error(error);
    res.status(500).json({ error: "Error obteniendo la tasa" });
  }
});

app.listen(3000, () => {
  console.log("Servidor webhook escuchando en puerto 3000");
});
